  2. const int trigPin = 52;
  3. const int echoPin = 53;
  4. const int buzzer = 50;
  5. const int ledPin = 51;
  6. // defines variables
  7. long duration;
  8. int distance;
  9. int safetyDistance;
  10. void setup() {
  11. pinMode(trigPin, OUTPUT); // Sets the trigPin as an Output
  12. pinMode(echoPin, INPUT); // Sets the echoPin as an Input
  13. pinMode(buzzer, OUTPUT);
  14. pinMode(ledPin, OUTPUT);
  15. Serial.begin(9600); // Starts the serial communication
  16. }
  17. void loop() {
  18. // Clears the trigPin
  19. digitalWrite(trigPin, LOW);
  20. delayMicroseconds(2);
  21. // Sets the trigPin on HIGH state for 10 micro seconds
  22. digitalWrite(trigPin, HIGH);
  23. delayMicroseconds(10);
  24. digitalWrite(trigPin, LOW);
  25. // Reads the echoPin, returns the sound wave travel time in microseconds
  26. duration = pulseIn(echoPin, HIGH);
  27. // Calculating the distance
  28. distance= duration*0.034/2;
  29. safetyDistance = distance;
  30. if (safetyDistance <= 5){
  31.  digitalWrite(buzzer, HIGH);
  32.  digitalWrite(ledPin, HIGH);
  33. }
  34. else{
  35.  digitalWrite(buzzer, LOW);
  36.  digitalWrite(ledPin, LOW);
  37. }
  38. // Prints the distance on the Serial Monitor
  39. Serial.print("Distance: ");
  40. Serial.println(distance);
  41. }
